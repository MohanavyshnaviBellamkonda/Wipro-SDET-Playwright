//PROTOTYPAL INHERITANCE :
Prototypal inheritance is the type of inheritance where objects inherits the properties from other objects using prototype.
Example:

function person(name) 
{
    this.name = name;
}
 
person.prototype.sayHi = function() 
{
    console.log('hello ' + this.name);
}
const me = new person("aryan");
const you = new person("vyshu");
me.sayHi();
you.sayHi();
console.log(Object.getPrototypeOf(me) === person.prototype) //true
console.log(me.sayHi === you.sayHi) // true
